### [661. Image Smoother](https://leetcode.com/problems/image-smoother/)
Kang
```java
class Solution {
    public int[][] imageSmoother(int[][] img) {
        int rows = img.length-1;
        int cols = img[0].length-1;
        int[][] newimg = new int[rows+1][cols+1];
        
        //For each row
        for(int i = 0 ; i <= rows ; i++){
            //For each col
            for(int j = 0; j <= cols ; j++){
                int[] filter = new int[9];
                
                filter[0] = img[i][j];
                
                if(i > 0){ //Add top
                    filter[1] = img[i-1][j];
                    
                    if(j > 0){ //Add top-left
                        filter[2] = img[i-1][j-1];
                    }else {
                    filter[2] = -1;
                    }
                    
                    if(j < cols){//Add top-right
                        filter[3] = img[i-1][j+1];
                    } else {
                        filter[3] = -1;
                    }
                } else {
                    filter[1] = -1;
                    filter[2] = -1;
                    filter[3] = -1;
                }
                
                if(i < rows){ //Add bot
                    filter[4] = img[i+1][j];
                    if( j > 0){ //Add bot-left
                        filter[5] = img[i+1][j-1];
                    }else {
                        filter[5] = -1;
                    }
                    if(j < cols){ //Add bot-right
                        filter[6] = img[i+1][j+1];
                    } else {
                        filter[6] = -1;
                    }
                } else {
                    filter[4] = -1;
                    filter[5] = -1;
                    filter[6] = -1;
                }
                
                if(j > 0){ //Add left
                    filter[7] = img[i][j-1];
                } else {
                    filter[7] = -1;
                }
                if(j < cols){
                    filter[8] = img[i][j+1];
                } else {
                    filter[8] = -1;
                }
                    
                int sum = 0;
                int count = 0;
                //Loop nums for average
                for(int n = 0; n < 9 ; n++){
                    int num = filter[n];
                    if(num >= 0){
                        sum+= num;
                        count++;
                    }
                }
                int avg = sum/count;
                newimg[i][j] = avg;
            }
        }
        return newimg;
    }
}
```
