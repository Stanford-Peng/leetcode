### 1. Surrounded Regions
 这道题有点逆向思维，先找到可以连到边界的O，标出来，剩下的就是被圈住的O
[Leetcode 130](https://leetcode.com/problems/surrounded-regions/)

Stan
```
    int m;
    int n;
    public void solve(char[][] board) {
        //circle the outer first
        this.m = board.length;
        this.n = board[0].length;
        //search from all outer elements
        for(int i = 0; i < m; i++){
            dfs(board,i,0);
            dfs(board,i,n-1);
        }
        for(int i = 0; i < n; i++){
            dfs(board,0,i);
            dfs(board,m-1,i);
        }
        
        //do the final changes
        for(int i = 0; i < m; i++){
            for(int j = 0; j < n; j++){
                if(board[i][j]=='A'){
                    board[i][j] = 'O';
                }else if(board[i][j]=='O'){
                     board[i][j] = 'X';
                }
            }
        }
    }
    
    public void dfs(char[][] board, int r, int c){
        //flip the o in the outer side to be another char 'a'
        if(r < 0 || r >= m || c < 0 || c >= n || board[r][c] != 'O'){
            return;
        }
        board[r][c] = 'A';
        dfs(board, r+1,c);
        dfs(board, r-1,c);
        dfs(board, r,c+1);
        dfs(board, r,c-1);
    }
```
### 2. Assign Cookies (Easy)
Guang [LeetCode](https://leetcode.com/problems/assign-cookies/description/)
```
class Solution{
    public int findContentChildren(int[] g, int[] s) {
      int lg = g.length;
      int ls = s.length;
      Arrays.sort(g);
      Arrays.sort(s);
      int children = 0;
      int cookie = 0; 
      while(children < lg && cookie < ls){
          if(s[cookie] >= g[children]){
             children++;
          }
          cookie++;
      }
      
   return children;   
}
}
```

### 3. Non-overlapping Intervals (Medium)
Guang [LeetCode](https://leetcode.com/problems/non-overlapping-intervals/description/)
```
class Solution {
    public int eraseOverlapIntervals(int[][] intervals) {
      //intervals里面的数组 按照第二个数字的大小 排序
     Arrays.sort(intervals,(a,b) -> a[1] - b[1]);
     int row = intervals.length;
     int current = 1; 
     int max = intervals[0][1]; 
     for(int i = 1; i < row;i++){
         if(intervals[i][0] < max){            
             continue;
         }
         current++;
         max = intervals[i][1];
     }
     return row - current;
    }
}
```

### 4. Pacific Atlantic Water Flow
[leetcode 417](https://leetcode.com/problems/pacific-atlantic-water-flow/)
Stan
当想记录递归多个结果，设全局变量。并注意已访问的节点避免endless loop。设置公共结束条件：

```
class Solution {
    int m;
    int n;
    boolean up = false;
    boolean left = false;
    boolean down = false;
    boolean right = false;
    public List<List<Integer>> pacificAtlantic(int[][] heights) {
        this.m = heights.length;
        this.n = heights[0].length;
        List<List<Integer>> res = new ArrayList<List<Integer>>();
        for(int i=0; i < m; i++){
            for(int j=0; j < n; j++){
                boolean[][] visited = new boolean[m][n];
                dfs(heights,i,j,visited);
                if((up||left)&&(right||down)){

                    ArrayList<Integer> temp = new ArrayList<>();
                    temp.add(i);
                    temp.add(j);
                    res.add(temp);
                    
                }
               up = false;
               left = false;
                down = false;
                right = false;
                
            }
        }
        return res;
    }
    public void dfs(int[][] heights, int r, int c, boolean[][] visited){
        if((up||left)&&(right||down)){
            return;
        }
        visited[r][c] = true;
        
        if(r == 0 ){
            up = true;
        }

        if(c == 0){
            left = true;
        }
        if(r == m-1 ){
            down = true;
        }
        if(c == n-1){
            right = true;
        }
 
        if( c >= 0 &&c < n &r+1 >=0 &&r+1 < m &&heights[r+1][c] <= heights[r][c] && !visited[r+1][c]){
            dfs(heights, r+1, c,visited);
            }
        if( c >= 0 &&c < n &r-1 >=0 &&r-1 < m &&heights[r-1][c] <= heights[r][c] && !visited[r-1][c]){
            dfs(heights, r-1, c,visited);
            }
        if( c+1 >= 0 &&c+1 < n &r >=0 &&r < m &&heights[r][c+1] <= heights[r][c] && !visited[r][c+1]){
            dfs(heights, r, c+1,visited);
            }
        if( c-1 >= 0 &&c-1 < n &r >=0 &&r < m &&heights[r][c-1] <= heights[r][c] && !visited[r][c-1]){
            dfs(heights, r, c-1,visited);
            }
        
    }
 }
```
