### Spiral Matrix

[leetcode](https://leetcode.com/problems/spiral-matrix/)

貌似面试笔试算法更喜欢考察基本的逻辑

```
class Solution {
    public List<Integer> spiralOrder(int[][] matrix) {
        int m = matrix.length;
        int n = matrix[0].length;

        int left = 0;
        int right = n-1;
        int top = 0;
        int bottom = m-1;
        
        List<Integer> res = new ArrayList<>();
        
        while(left<=right && top<=bottom){

            for (int i = left; i <= right ; i++) {
                res.add(matrix[top][i]);
            }
            top++;
            if(top > bottom) break;

            for (int i = top; i <= bottom; i++) {
                res.add(matrix[i][right]);
            }
            right--;
            if(right < left) break;
            for (int i = right; i >= left; i--) {
                res.add(matrix[bottom][i]);
            }
            bottom--;
            if(bottom < top) break;
            for (int i = bottom; i >= top; i--) {
                res.add(matrix[i][left]);
            }
            left++;
            if(left>right) break;
        }
        return res;
    }
}
```

```
class Solution {
    public int findInMountainArray(int target, MountainArray mountainArr) {
        int l = 0, r = mountainArr.length() - 1;
        while (l < r) {
            int mid = (l + r) / 2;
            if (mountainArr.get(mid) < mountainArr.get(mid + 1)) {
                l = mid + 1;
            } else {
                r = mid;
            }
        }
        int peak = l;
        int index = binarySearch(mountainArr, target, 0, peak, true);
        if (index != -1) {
            return index;
        }
        return binarySearch(mountainArr, target, peak + 1, mountainArr.length() - 1, false);
    }

    public int binarySearch(MountainArray mountainArr, int target, int l, int r, boolean flag) {
        if (!flag) {
            target *= -1;
        }
        while (l <= r) {
            int mid = (l + r) / 2;
            
            int cur = mountainArr.get(mid)* (flag ? 1 : -1);// * (flag ? 1 : -1)
            System.out.println(cur);
            if (cur == target) {
                return mid;
            } else if (cur < target) {
                l = mid + 1;
            } else {
                r = mid - 1;
            }
        }
        return -1;
    }
}
```

### 连续子数组的最大和
Guang 
```
class Solution {
    public int maxSubArray(int[] nums) {
       int pro = nums[0];
       int length = nums.length;
       int max = nums[0]; 
       for(int i = 1; i < length; i++){
           pro = Math.max(pro+nums[i],nums[i]);
           max = Math.max(pro,max);
       }

       return max; 
    }
}
```

###  最小K个数
Guang
```
class Solution {
    public int[] smallestK(int[] arr, int k) {
       int length = arr.length;
       return quickSort(arr,0,length - 1, k);
    }
    
    public int[] quickSort(int[] arr,int l,int r,int k){
       if(l >= r){
         return new int[0];
       } 
       int value = arr[l];
       int left = l;
       int right = r; 
       while(l < r){
           while(value > arr[l]){
            l++;
           }
           while(value < arr[r]){
               r--;
           }
           //swap
           if(l < r){
             int temp = arr[l];
             arr[l] = arr[r];
             arr[r] = temp;
           }

       }
       if(r + 1 < k ){
        quickSort(arr,r + 1,right,k);
       }else if(r + 1 > k){
           quickSort(arr,left,r,k);
       }
       return Arrays.copyOf(arr,k);
    }

}
```
