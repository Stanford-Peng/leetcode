### 1. 最大联通面积
Stan
[leetcode 695](https://leetcode.com/problems/max-area-of-island/)
```
    int[][] grid;
    int m;
    int n;
    int[][] directions = new int[][]{{1,0},{-1,0},{0,1},{0,-1}};
    public int maxAreaOfIsland(int[][] grid) {
        if (grid == null || grid.length == 0) {
            return 0;
        }
        this.grid = grid;
        int m = grid.length;
        int n = grid[0].length;
        this.m=m;
        this.n=n;
        int res = 0;
        for(int i = 0;i < m;i++){
            for(int j = 0;j<n;j++){
                res = Math.max(dfs(i,j),res);
            }
        }
        return res;
    }
    
    int dfs(int x, int y){
        if( x < 0 || x >=m || y < 0|| y >= n || grid[x][y] == 0 ){
            return 0;
        } 
        int res = 1;
        grid[x][y] = 0;//essential marked as visited otherwise stackoverflow
        for(int[] d : directions){
            int curx = x+d[0];
            int cury = y+d[1];
            res+=dfs(curx,cury);
        }
        return res;
    }

### 3.最多联通数
Stan
[leetcode 200](https://leetcode.com/problems/number-of-islands/submissions/)
```
class Solution {
    
    char[][] grid;
    int[][] directions = new int[][]{
        {1,0},
        {-1,0},
        {0,1},
        {0,-1}
    };
    int m;
    int n;
    
    public int numIslands(char[][] grid) {
        this.m = grid.length;
        this.n = grid[0].length;
        this.grid = grid;
        int res = 0;
        for(int i = 0;i < m;i++){
            for(int j = 0;j < n;j++){
                if(grid[i][j] == '1'){
                    res++;
                    dfs(i,j);
                }
            }
        }
        return res;
    }
    
    void dfs(int i, int j){
        if(i < 0 || j < 0 || i>=m||j>=n||grid[i][j]=='0'){
            return;
        }
        grid[i][j] = '0';
        for(int[] d : directions){
            dfs(i+d[1],j+d[0]);
        }
    }
}
