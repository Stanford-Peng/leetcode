// 012
// 345 
var slidingPuzzle = function(board) {
    const nei = [[1, 3], [0, 2, 4], [1, 5], [0, 4], [1, 3, 5], [2, 4]];
    const  charArray = []
    for(let i = 0; i < 2;i++){
        for(let j = 0; j < 3;j++){
            charArray.push(board[i][j])
        }
    }
    //to String 
    const boardString = charArray.join("")
    if(boardString === "123450"){
        return 0;
    }
    //enumerate
    const get = (input) =>{
       const result = []
       const array = Array.from(input)
       const x = input.indexOf('0')
       for(const y of nei[x]){
        const temp = array[x]
         array[x] = array[y]
         array[y] = temp
         result.push(array.join(""))
         const temp1 = array[x]
         array[x] = array[y]
         array[y] = temp1
       } 
      return result
    }
    const queue = []
    let step = 0
    queue.push(boardString)
    const passed = new Set()
    passed.add(boardString)

    while(queue.length){
       step++ 
       const size = queue.length
       for(let i = 0; i < size;i++){
          const current = queue.shift()
          for(const next of get(current)){
              if(!passed.has(next)){
                 if(next === "123450"){
                     return step
                 }
                 queue.push(next)
                 passed.add(next)
              }
             
          }
       }
    }
  return -1;  
};
