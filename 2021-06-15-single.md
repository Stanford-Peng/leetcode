### [Peak Index in a Mountain Array](https://leetcode.com/problems/peak-index-in-a-mountain-array/)
Guang
```java
class Solution {
    public int peakIndexInMountainArray(int[] arr) {
        int length = arr.length;
        if(length < 3){
            return 0;
        }
        int l = 0;
        int r = length - 1; 
        while(l < r){
            int mid = l + (r - l) / 2;
            if(arr[mid] < arr[mid - 1]){
              r = mid;
            }else {
              l = mid + 1; 
            }
        }
     return l - 1;
    }
}
```
