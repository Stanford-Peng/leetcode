```
class Solution {
    int length;
    List<List<Integer>> res;
    
    public List<List<Integer>> subsets(int[] nums) {
        length = nums.length;
        List<Integer> prefix = new ArrayList<>();
        res = new ArrayList<List<Integer>>();
        
        backtrack(prefix, nums,0);
        return res;
        
    }
    
    void backtrack(List<Integer> prefix, int[] nums, int start){
        
        if(prefix.size() <= length){
            res.add(new ArrayList<>(prefix));
        }else{
            return;
        }
        
        for(int i = start; i < length; i++){
            prefix.add(nums[i]);
            backtrack(prefix, nums, i+1);
            prefix.remove(prefix.size()-1);
        }
        
        
        
    }
}
```

```
class Solution {
    int length;
    List<List<Integer>> res;
    public List<List<Integer>> subsetsWithDup(int[] nums) {
        length = nums.length;
        
        res = new ArrayList<>();
        List<Integer> prefix = new ArrayList<>();
        
        Arrays.sort(nums);
        backtrack(prefix, nums, 0);
        return res;
    }
    
    void backtrack(List<Integer> prefix, int[] nums, int start){
        

        if(prefix.size() < length){
            res.add(new ArrayList<>(prefix));
        }else if(prefix.size() == length){
            res.add(new ArrayList<>(prefix));
            return;
        }
        
        int pre = 0;
        for(int i = start;i < length;i++){
            if(pre != 0 && nums[i] == pre) continue;
            prefix.add(nums[i]);
            pre = nums[i];
            backtrack(prefix,nums, i+1);
            prefix.remove(prefix.size()-1);
            
        }
    }
}
```

```java
class Solution {
    public int sumBase(int n, int k) {
        int num = 0;
        int quotient = n;
        int remainder = 0;
        int i = 0;
        while(quotient !=0){
            remainder = quotient%k;
            quotient = quotient/k;
            num = (int)(remainder * Math.pow(10,i)) + num;
            i++;
        }
        int res = 0;
        while(num/10 > 0){
            res += num%10;
            num /= 10;
        }
        res += num%10;
        return res;
    }
```

### [942. DI String Match](https://leetcode.com/problems/di-string-match/)
Kang - Math
```java
class Solution {
    public int[] diStringMatch(String s) {
        char[] c = s.toCharArray();
        int len = c.length;
        int[] res = new int[len+1];
        int ceil = len;
        int floor = 0;
        for (int i = 0; i < len; i++) {
            if(c[i] =='I'){
                res[i] = floor++;
            } else {
                res[i] = ceil--;
            }
        }
        res[len] = c[len-1] == 'I' ? floor : ceil;
        return res;
    }
}
```

### [927 · 翻转字符串II](https://www.lintcode.com/problem/927/?_from=enterprise&fromId=20 "927 · 翻转字符串II")

```java
public char[] reverseWords(char[] str) {
	StringBuilder builder = new StringBuilder("");
	boolean flag = false;
	for (int i = 0; i < str.length; i++) {
		if(str[i] == ' '){
			flag = true;
			continue;
		}
		if(flag){
			if(builder.length() != 0){
				builder.append(' ');
			}
			flag = false;
		}
		builder.append(str[i]);
	}
	char[] res = new char[builder.length()];
	String[] split = builder.toString().split(" ");
	int index = res.length-1;
	for (String s : split) {
		for (int i = s.length()-1; i >= 0; i--) {
			res[index] = s.charAt(i);
			index--;
		}
		if(index <= 0){
			break;
		}
		res[index--] = ' ';
	}
	return res;
}
```
