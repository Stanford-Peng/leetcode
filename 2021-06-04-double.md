### Longest Harmonious Subsequence

[leetcode 594](https://leetcode.com/problems/longest-harmonious-subsequence/submissions/)

Stan

```
class Solution {
    public int findLHS(int[] nums) {
        Map<Integer, Integer> map = new TreeMap<>();
        for(int i:nums){
            int frequency = map.getOrDefault(i,0);
            frequency++;
            map.put(i,frequency);
        }
        
        int max = 0;
        for(Map.Entry<Integer,Integer> e: map.entrySet()){
            if(map.containsKey(e.getKey()+1))
                max = Math.max(max,map.get(e.getKey()+1)+e.getValue());
        }
        
        return max;

    }
}
```

```
class Solution {
    public int longestConsecutive(int[] nums) {
        if(nums.length == 0) return 0;
        Arrays.sort(nums);
        int length = 1;
        int max = 0;

        int pre = nums[0];
        for(int i = 1;i < nums.length;i++){
            int cur = nums[i];
            if(pre == cur) continue;
            if(cur - pre == 1){
                length++;
            }else{
                max = Math.max(max,length);
                length =1;
            }
            pre = cur;
            
        }
        max = Math.max(max,length);
        return max;
            
    }
}
```

```
public int longestConsecutive(int[] nums) {
    Map<Integer, Integer> countForNum = new HashMap<>();
    for (int num : nums) {
        countForNum.put(num, 1);
    }
    for (int num : nums) {
        forward(countForNum, num);
    }
    return maxCount(countForNum);
}

private int forward(Map<Integer, Integer> countForNum, int num) {
    if (!countForNum.containsKey(num)) {
        return 0;
    }
    int cnt = countForNum.get(num);
    if (cnt > 1) {
        return cnt;
    }
    cnt = forward(countForNum, num + 1) + 1;
    countForNum.put(num, cnt);
    return cnt;
}

private int maxCount(Map<Integer, Integer> countForNum) {
    int max = 0;
    for (int num : countForNum.keySet()) {
        max = Math.max(max, countForNum.get(num));
    }
    return max;
}
```

### [Number of Islands](https://leetcode.com/problems/number-of-islands/)
Guang BFS
```java
class Solution {
    public int numIslands(char[][] grid) {
       int m = grid.length;
       int n = grid[0].length;
       int num = 0; 
       for(int i = 0; i < m; i++){
            for(int j = 0; j < n; j++){
              if(grid[i][j] == '1'){
                   bfs(grid,i,j,m,n);
                   num++; 
              }
            }
       }
       return num;
    }
     
    public void  bfs(char[][] grid,int i, int j, int m, int n){
       Queue<int[]> queue = new LinkedList<>();
       queue.offer(new int[] {i,j}); 
       while(!queue.isEmpty()){
          int[] temp = queue.poll(); 
          i = temp[0]; j = temp[1]; 
          if((0 <= i && i < m )&&(0 <= j && j < n ) && grid[i][j] == '1' ){
            grid[i][j] = '0';
            if(i - 1 >= 0 )
               queue.offer(new int[] {i-1,j});
            if(j - 1 >= 0)
               queue.offer(new int[] {i,j-1});
            if(i + 1 <= m)
               queue.offer(new int[]{i + 1, j});
            if(j + 1 <= n)
               queue.offer(new int[]{i,j + 1});
          }
       }
    } 
}
```

### [Binary Tree Right Side View](https://leetcode.com/problems/binary-tree-right-side-view/)
Guang BFS
```java
/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    public List<Integer> rightSideView(TreeNode root) {
      if(root == null)
        return new ArrayList<>();  
      Queue<TreeNode> queue = new LinkedList<>(); 
      queue.offer(root);  
      List<Integer> list = new ArrayList<>(); 
      while(!queue.isEmpty()){
         int size = queue.size(); 
         for(int i = 0; i < size; i++){
             TreeNode temp = queue.poll(); 
             if(i == size - 1)
               list.add(temp.val);
             if(temp.left != null)
               queue.offer(temp.left);
             if(temp.right != null)
               queue.offer(temp.right);              
         }
      }
      return list;
    }
}
```

### [1160. Find Words That Can Be Formed by Characters](https://leetcode.com/problems/find-words-that-can-be-formed-by-characters/)
Kang - Array
```java
class Solution {
    public int countCharacters(String[] words, String chars) {
        StringBuilder sb;
        
        int wordlen = words.length;
        int res = 0;
        
        for(int i = 0; i < wordlen; i++){
            String word = words[i];
            int wlen = word.length();
            sb = new StringBuilder(chars);
            boolean found = true;
            for(int j = 0; j < wlen; j++){
                int index = sb.indexOf(String.valueOf(word.charAt(j)));
                if(index != -1){
                    sb.deleteCharAt(index);
                } else{
                    found = false;
                    break;
                }
            }
            if(found){
                res += wlen;
            }
            
        }
        return res;
    }
}
```

### [509. Fibonacci Number](https://leetcode.com/problems/fibonacci-number/)
Kang - Array
```java
class Solution {
    
    public int fib(int n) {
        int counter = 0;
        int[] map = new int[31];
        map[0] = 0;
        map[1] = 1;
        map[2] = 1;
        if(n == 0 || n == 1 || n == 2){
            return map[n];
        }
        counter = 2;
        while(counter != n){
            counter++;
            int next = map[counter-1] + map[counter-2];
            map[counter] = next;
        }
        return map[counter];
    }
}
```
