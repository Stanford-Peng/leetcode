### [First Bad Version (Easy)](https://leetcode.com/problems/first-bad-version/description/)
Guang 
```
/* The isBadVersion API is defined in the parent class VersionControl.
      boolean isBadVersion(int version); */

public class Solution extends VersionControl {
    public int firstBadVersion(int n) {
       int l = 1;
       int r = n; 
       while(l < r){
         int mid = l + (r - l) / 2;
         if(isBadVersion(mid)){
            r = mid;
         }else{
            l = mid + 1;
         }
       }
    
      return l; 
    }
}

### [Single Element in a Sorted Array (Medium)](https://leetcode.com/problems/single-element-in-a-sorted-array/description/)
Guang 
```
class Solution {
    public int singleNonDuplicate(int[] nums) {
       int l = 0;
       int r = nums.length - 1;
        while(l < r){
            int mid = l + (r - l) / 2;
            if(mid % 2 == 1){
                --mid;
            } 
            if(nums[mid] == nums[mid + 1]){
               l = mid + 2;
            }else{
                 r = mid;
            }

        }      

       return nums[r];
    }
}

class Solution {
    public int[] dailyTemperatures(int[] temperatures) {
        Stack<Integer> stack = new Stack<>();
        // stack.push(temperatures[0]);
        int length = temperatures.length;
        int[] res = new int[length];
        // int count = 0;
        // int index = 0;
        for(int i = 0;i < length;i++){
            // count++;
            // int 
            while(stack.size() > 0 && temperatures[i] > temperatures[stack.peek()]){
                    int pre = stack.pop();
                    res[pre] = i-pre;
                    // stack.pop()
                    // count++;
                    // index++;
            }
            stack.push(i);
        }
        
        return res;
        
    }
}
